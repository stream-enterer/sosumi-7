```rust
//! F010 H1 fix verification: Clear records → replays → fills pixels.
//!
//! Per spec docs/superpowers/specs/2026-04-26-F010-h1-fix-design.md (Test 2).
//!
//! Pre-fix: this test fails to compile (DrawOp::Clear doesn't exist) or fails
//! at the replay-pixel assertion (Clear records nothing, so the replayed image
//! retains its sentinel pre-fill).
//!
//! Post-fix: passes — recording captures DrawOp::Clear { color }, replay paints
//! the painter's current clip rect with the requested color.

use emcore::emColor::emColor;
use emcore::emImage::emImage;
use emcore::emPainter::emPainter;
use emcore::emPainterDrawList::{DrawOp, RecordedOp, replay};

#[test]
fn f010_h1_clear_records_and_replays_pixels() {
    // ---- Phase 1: record ----
    let mut ops: Vec<RecordedOp> = Vec::new();
    let target_color = emColor::rgba(200, 100, 50, 255);
    {
        let mut recorder = emPainter::new_recording(800, 600, &mut ops);
        recorder.SetClipping(50.0, 40.0, 750.0, 540.0);
        recorder.Clear(target_color);
    }

    // Recording must capture exactly one DrawOp::Clear with the passed color.
    let clear_ops: Vec<&RecordedOp> = ops
        .iter()
        .filter(|r| matches!(r.op, DrawOp::Clear { .. }))
        .collect();
    assert_eq!(
        clear_ops.len(),
        1,
        "expected exactly 1 DrawOp::Clear, got {} (ops: {:?})",
        clear_ops.len(),
        ops
    );
    match &clear_ops[0].op {
        DrawOp::Clear { color } => {
            assert_eq!(*color, target_color, "recorded color mismatches");
        }
        _ => unreachable!(),
    }

    // ---- Phase 2: replay into a sentinel-pre-filled emImage ----
    let mut img = emImage::new(800, 600, 4);
    let sentinel = emColor::rgba(0, 0, 0, 0);
    img.Fill(0, 0, 800, 600, sentinel.GetRed(), sentinel.GetGreen(), sentinel.GetBlue(), sentinel.GetAlpha());

    {
        let mut replay_painter = emPainter::new(&mut img);
        replay_painter.SetClipping(50.0, 40.0, 750.0, 540.0);
        replay(&ops, &mut replay_painter);
    }

    // ---- Phase 3: assert pixels ----
    // Inside the clip: target_color. Outside the clip: still sentinel.
    let inside_samples = [(400, 300), (51, 41), (749, 539)];
    for (x, y) in inside_samples {
        let actual = img.GetPixel(x, y);
        assert_eq!(
            actual, target_color,
            "pixel ({}, {}) expected {:?}, got {:?}",
            x, y, target_color, actual
        );
    }
    let outside_samples = [(10, 10), (790, 590), (0, 0)];
    for (x, y) in outside_samples {
        let actual = img.GetPixel(x, y);
        assert_eq!(
            actual, sentinel,
            "pixel ({}, {}) outside clip — expected sentinel {:?}, got {:?}",
            x, y, sentinel, actual
        );
    }
}
```

use emcore::emColor::emColor;
use emcore::emPainter::emPainter;
use emcore::emPainterDrawList::{DrawOp, RecordedOp};

#[test]

    let mut ops: Vec<RecordedOp> = Vec::new();
    let target_color = emColor::rgba(255, 0, 0, 255);
    {
        let mut painter = emPainter::new_recording(800, 600, &mut ops);
        painter.Clear(target_color);

    let observation_artifact = serde_json::json!({

        "fix_landed": true,
    });

        .join("../../docs/debug/investigations/F010-investigation/artifacts/H1-ops.json");
    std::fs::create_dir_all(path.parent().unwrap()).unwrap();
    std::fs::write(&path, observation_artifact.to_string()).unwrap();

    assert_eq!(
        len_after, 1,
        "post-fix: Clear must record exactly one op, got {}",
        len_after
    );
    match &ops[0].op {
        DrawOp::Clear { color } => {
            assert_eq!(*color, target_color, "recorded color mismatches caller's");
        }
        other => panic!("expected DrawOp::Clear, got {:?}", other),

}
